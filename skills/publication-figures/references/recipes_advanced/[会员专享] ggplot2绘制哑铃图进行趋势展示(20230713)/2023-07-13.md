### 数据清洗
```

df1 <- read_tsv("data-1.xls")

df <- read_tsv("data.xls") %>%  # 读取名为 data.xls 的 TSV 文件，并将数据存储在 df 变量中
  filter(!is.na(Raised)) %>%  # 过滤掉 Raised 列中的缺失值
  select(Year, Accepted, Raised) %>%  # 选择 Year、Accepted 和 Raised 列
  inner_join(df1, by = "Year") %>%  # 与 winners 数据集进行内连接，根据 Year 列进行匹配
  arrange(Year) %>%  # 按照 Year 列进行升序排列
  mutate(Nationality = as.factor(Nationality)) %>%  # 将 Nationality 列转换为因子（factor）类型
  filter(Nationality %in% c("Kenya", "United Kingdom", "United States")) %>%  # 过滤出 Nationality 列为 "Kenya"、"United Kingdom" 和 "United States" 的行
  mutate(Accepted = scale(Accepted, center = FALSE, scale = TRUE),  # 对 Accepted 列进行标准化
         Raised = scale(Raised, center = FALSE, scale = TRUE))  # 对 Raised 列进行标准化
```

### 数据可视化
```r
df %>% ggplot(aes(x = Year)) +  # 创建一个 ggplot 对象，设置 x 轴的数据为 Year 列
  geom_point(aes(y = Accepted),  # 添加散点图层，设置 y 轴的数据为 Accepted 列
             shape = 21, stroke = 0.5, size = 3,
             color = "black", fill = "#788FCE", key_glyph = draw_key_rect) +  # 设置散点的形状、描边、大小、颜色和填充色
  geom_point(aes(y = Raised),  # 添加散点图层，设置 y 轴的数据为 Raised 列
             shape = 21, stroke = 0.5, size = 3, fill = "#E6956F",
             color = "black", key_glyph = draw_key_rect) +  # 设置散点的形状、描边、大小、颜色和填充色
  geom_line(aes(y = Accepted, color = "Accepted"), linewidth = 1, key_glyph = draw_key_rect) +  # 添加线条图层，设置 y 轴的数据为 Accepted 列，颜色为 "Accepted"
  geom_line(aes(y = Raised, color = "Raised"), linewidth = 1, key_glyph = draw_key_rect) +  # 添加线条图层，设置 y 轴的数据为 Raised 列，颜色为 "Raised"
  geom_segment(aes(xend = Year, y = Accepted, yend = Raised), color = "black") +  # 添加线段图层，连接 Accepted 和 Raised 列的数据
  facet_wrap2(~Nationality, strip = ridiculous_strips) +  # 按 Nationality 列进行分面
  scale_x_continuous(breaks = c(2007, 2010, 2014, 2017)) +  # 设置 x 轴刻度线的位置
  scale_colour_manual("", breaks = c("Accepted", "Raised"),  # 设置颜色图例的标签和颜色
                      values = c("#788FCE", "#E6956F")) +
  guides(fill = "none") +  # 隐藏填充色图例
  theme_minimal() +  # 设置主题为 minimal
  theme(text = element_text(size = 10, color = "black", face = "bold"),  # 设置文本样式
        strip.background = element_rect(color = "#788FCE", fill = "#788FCE"),  # 设置分面背景的颜色
        strip.text = element_text(color = "white", size = 10, face = "bold"),  # 设置分面文本的样式
        panel.spacing = unit(0, "cm"),  # 设置面板间距为 0
        legend.position = "top",  # 将图例置于顶部
        axis.title = element_blank(),  # 隐藏坐标轴标题
        plot.margin = margin(0, 10, 10, 10))  # 设置绘图区的边距
```
