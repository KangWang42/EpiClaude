### 加载R包
```r
pacman::p_load(tidyverse, hrbrthemes, ggfittext,ggtext,scales,ggtext)
```
### 导入数据
```python
data <- readr::read_csv('product_hunt.csv')
```
### 数据清洗
```python
df <- data %>%  # 选择'upvotes'和'category_tags'列
  select(upvotes, category_tags) %>% 
  # 将'category_tags'列中的多个标签分开
  separate_rows(category_tags, sep = ", ") %>% 
  # 去除标签中的特殊字符
  mutate(name_2 = gsub("[^[:alnum:]]", " ",category_tags)) %>% 
  # 去除标签两端的空白字符
  mutate(name_2 = trimws(name_2)) %>% 
  # 按'name_2'进行分组，并计算每组的数量
  group_by(name_2) %>% summarize(total=n()) %>% 
  # 计算每个标签的百分比
  mutate(percentage = total/sum(total),
         percentage=percent(percentage,accuracy = 0.01)) %>% 
  top_n(5) %>% # 选择前5个最常见的标签
  arrange(desc(total))   # 按'total'降序排列

df <- df %>% arrange(desc(total)) %>% select(2,1,3) %>%
  mutate(x_axis = rep(5, each = 10, length = n()),
         x_axis_2 = x_axis + 1.2,
         x_axis_3 = x_axis + 2.4,
         y_axis = rep(-1:-10, length = n()))
```
### 定义因子
```python
df$name_2 <- fct_relevel(df$name_2, c("ANDROID","DEVELOPER TOOLS","WEB APP",
                                      "IPHONE","PRODUCTIVITY"))
```
