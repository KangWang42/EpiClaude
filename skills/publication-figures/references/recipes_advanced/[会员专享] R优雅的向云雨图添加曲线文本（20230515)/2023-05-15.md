### 数据清洗
```r
dat <- df %>% 
  group_by(boardgamecategory) %>% # 按游戏类型分组
  summarise(dens_x = density(average)$x, # 计算平均分数的密度函数的横坐标
            dens_y = density(average)$y, # 计算平均分数的密度函数的纵坐标
            .groups = 'drop') %>% # 取消分组
  mutate(dens_y = dens_y / max(dens_y) * 0.8, # 将密度函数规范化到 [0, 0.8] 区间
         dens_y = dens_y + case_when( # 根据游戏类型调整密度函数的位置
           boardgamecategory == 'Card Game' ~ 1, 
           boardgamecategory == 'Dice' ~ 2,
           boardgamecategory == 'Fantasy' ~ 3,
           boardgamecategory == 'Party Game' ~ 0,
           boardgamecategory == 'Wargame' ~ 4)) %>% 
  filter(dens_x > 3, (dens_x > 5 | boardgamecategory != 'Wargame')) # 过滤密度函数横坐标小于3，或者横坐标大于5且游戏类型不是 Wargame 的数据点
```

### 绘制云雨图
```r
dat %>% 
  ggplot(aes(x = dens_x, y = dens_y, fill = boardgamecategory)) + # 设置散点图的横轴、纵轴和颜色映射
  geom_textpath(aes(col = boardgamecategory, label = boardgamecategory, # 添加文字标签，并设置文字颜色和内容
                    hjust = case_when(
                      boardgamecategory == 'Card Game' ~ 0.2, 
                      boardgamecategory == 'Dice' ~ 0.3,
                      boardgamecategory == 'Fantasy' ~ 0.35,
                      boardgamecategory == 'Party Game' ~ 0.2,
                      boardgamecategory == 'Wargame' ~ 0.15)), 
                vjust = -0.2, text_only = T,size = 4)+ # 设置文字的位置和大小
  geom_polygon() + # 绘制多边形
  geom_boxplot(data = df, # 添加箱线图
               aes(x = average, y = case_when(
                 boardgamecategory == 'Card Game' ~ 1, 
                 boardgamecategory == 'Dice' ~ 2,
                 boardgamecategory == 'Fantasy' ~ 3,
                 boardgamecategory == 'Party Game' ~ 0,
                 boardgamecategory == 'Wargame' ~ 4)),
               size = 1, width = 0.2, outlier.shape = NA)+
  scale_y_continuous(breaks = NULL, minor_breaks = NULL) +
  labs(x=NULL,y=NULL)+
  scale_fill_jco()+
  scale_color_jco()+
  theme_minimal() +
  theme(legend.position = 'none',
        text = element_text(size = 10,color="black",face="bold")) 
```
