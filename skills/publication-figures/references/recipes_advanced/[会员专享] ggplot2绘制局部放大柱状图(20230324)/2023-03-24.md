### 加载R包
```r
library(tidyverse)
library(scales)
library(patchwork)
library(ggsci)
```

### 导入数据
```r
df1 <- read_tsv("data1.xls")
# 将 type2 列转换为因子变量，并调整因子水平顺序为逆序
df1$type2 <- factor(df1$type2, levels = df1$type2 %>% rev())
```
### 绘制左图
```r
plot1 <- df1 %>% ggplot(aes(axis_label, total)) +  # 设置绘图参数
  geom_col(aes(fill = type2), width = 1) +  # 绘制柱状图，设置分组填充参数和柱宽
  geom_text(aes(x =0.2, y = textx , label = type2), hjust = 1, vjust = 1, size = 3) +  # 添加标签，设置位置、文本和样式参数
  scale_fill_npg() +  # 调整填充颜色
  scale_x_discrete(expand = expansion(mult = c(5, 0))) +  # 调整 x 轴跨度
  scale_y_continuous(limits = c(0,700000)) +  # 调整 y 轴范围
  guides(fill = FALSE) +  # 隐藏填充图例
  theme_void() +  # 设置无背景主题
  theme(axis.text.x = element_text(color="black", face="bold", size = 10))  # 调整 x 轴文本样式
```

### 绘图放大阴影
```r
poly <- tibble(x = c(0, 2, 2, 0),
               y = c(393153, 0, 560000, 506811))
```
```r
zoom <- ggplot() + # 创建 ggplot 对象
  # 绘制多边形，设置数据源和 x、y 坐标参数，填充颜色和透明度
  geom_polygon(data = poly, aes(x, y), fill = "#41CFA0", alpha = 0.3) +
  # 添加线段注释，设置起点和终点坐标，线段颜色
  annotate("segment", x = 0, xend = 2, y = 393153, yend = 0, color = "#41CFA0") +
  annotate("segment", x = 0, xend = 2, y = 506811, yend = 560000, color = "#41CFA0") +
  scale_y_continuous(limits = c(0, 700000)) +   # 调整 y 轴范围
  scale_x_continuous(expand = expansion(mult = c(0, 0))) +   # 调整 x 轴范围
  guides(fill = FALSE) +   # 隐藏填充图例
  theme_void()   # 设置无背景主题
```
### 绘制右图
```r
df2 <- read_tsv("data2.xls")
# 将 type 列转换为因子变量，并调整因子水平顺序为逆序
df2$type <- factor(df2$type, levels = df2$type %>% rev())  
```
```r
plot2 <- df2 %>% ggplot(aes(axis_label, production)) +  # 设置绘图参数
  geom_col(aes(fill = type), width = 1) +  # 绘制柱状图，设置分组填充参数和柱宽
  geom_segment(aes(x = 1.5, xend = 2, y = posy, yend = posy), size = 0.3) +  # 添加辅助线段，设置起点和终点坐标，线段粗细
  geom_segment(aes(x = 2, xend = 2.3, y = posy, yend = posyend), linetype = "12", size = 0.3) +  # 添加辅助线段，设置起点和终点坐标，线段类型和粗细
  geom_segment(aes(x = 2.3, xend = 4, y = posyend, yend = posyend), linetype = "12", size = 0.3) +  # 添加辅助线段，设置起点和终点坐标，线段类型和粗细
  geom_text(aes(x = 4.1, y = posyend, label = label), hjust = 0, size = 3) +  # 添加标签，设置位置、文本和样式参数
  geom_text(aes(x = 2.4, y = textx, label = type), hjust = 0, size = 3) +  # 添加标签，设置位置、文本和样式参数
  scale_fill_jama() +  # 调整填充颜色
  scale_x_discrete(expand = expansion(mult = c(0, 5))) +  # 调整 x 轴跨度
  guides(fill = FALSE) +  # 隐藏填充图例
  theme_void() +  # 设置无背景主题
  theme(axis.text.x = element_text(color = "#183170", size = 10))  # 调整 x 轴文本样式
```
