### 加载R包
```r
library(tidyverse)
library(glue)
library(ggtext)
library(janitor)
```
### 导入数据
```r
df <- readr::read_csv('data.txt')
```
### 数据清洗

```r
d <- df %>% janitor::clean_names() %>% # 将列名转换成小写并用下划线代替空格
  select(year, j_d) %>% drop_na()

anom <- tibble(j_d = seq(-.50, 1.25, .50)) %>% 
  # 对 j_d 列的值进行格式化，保留一位小数，并使用 "{x}°C" 样式进行格式化，小数点使用 "."，千位分隔符使用 ","
  mutate(label = gt::vec_fmt_number(j_d, decimals = 1,
                                    pattern = "{x}°C", sep_mark = ".", dec_mark = ","))
```

### 数据可视化
```r
ggplot(data = d, aes(x = year, y = j_d)) +
  geom_raster(aes(y = 1, fill = j_d)) +  # 添加 geom_raster 图层，y 值为 1，用 j_d 列的值填充图层
  geom_raster(aes(y = 0, fill = j_d)) +
  geom_raster(aes(y = -1, fill = j_d)) +
  # 添加灰色虚线水平线图层，y 值由 anom 数据框的 j_d 列决定
  geom_hline(yintercept = anom$j_d, color = "grey70", linetype = 3, linewidth = .3) +
  # 添加红色文本图层，标记 anom 数据框中的 j_d 列在 x=1880 处，文本内容为 label 列
  geom_text(
    data = anom, aes(x = 1880, y = j_d, label = label), inherit.aes = FALSE,
    angle = 0, hjust = 0, vjust = 0, color = "red", size = 4) +
  # 添加三个白色、灰色和黑色线条图层，线条粗细分别为 0.5、0.8 和 0.25，线条末端为 round
  geom_line(linewidth = 0.5, color = "white", lineend = "round") +
  geom_line(linewidth = 0.8, color = "grey70", lineend = "round") +
  geom_line(linewidth = .25, color = "black", lineend = "round") +
  scale_fill_gradientn(colours = rev(RColorBrewer::brewer.pal(11, "RdBu"))) + 
  scale_x_continuous(breaks = seq(1800, 2020, 20)) +   # 设置 x 轴的刻度间隔为每 20 年
  labs(x = NULL, y = NULL) +   # 去除 x 和 y 轴的标签
  # 设置坐标系为笛卡尔坐标系，y 轴范围为 -.55 到 1.1，不添加额外空白，裁剪超出边界的元素
  coord_cartesian(expand = FALSE, ylim = c(-.55, 1.1), clip = "on") +
  theme_void() +
  # 设置图表的纵横比为 0.5，图表的上下左右边距分别为 10、20、10、20
  theme(
    aspect.ratio = 0.5,
    plot.margin = margin(10, 20, 10, 20),
    # 设置 x 轴标签文本的颜色为黑色，水平和垂直居中对齐，字号为 10，加粗字体
    axis.text.x = element_text(color = "black", hjust = 0.5, vjust = 0.5, size = 10, face = "bold"),
    axis.ticks.x = element_line(color = "black", linewidth = .75),     # 设置 x 轴刻度线颜色为黑色，粗细为 0.75
    axis.ticks.length.x = unit(.2, "line"),     # 设置 x 轴刻度线长度为 0.2 个线段单位
    legend.position = "top",     # 设置图例在图表的上方
    legend.background = element_blank(),    # 去除图例背景和标题
    legend.title = element_blank(),
    panel.grid = element_blank()     # 去除网格线
  ) +
  # 设置填充颜色图例的方向为水平，不反转颜色条，颜色条宽度为 16cm，高度为 0.5cm
  guides(fill = guide_colorbar(direction = "horizontal", reverse = F, barwidth = unit(16, "cm"),
                               barheight = unit(0.5, "cm")))

```
