### 构建点文件
```r
nodes <- distinct(edges, name = to, group = from, size) %>% 
# 选取to、from和size三列，去除重复值，存入nodes变量中
is_main = group == "origin", # 设置是否为主节点
label = ifelse(is_main | size > 1000, name, NA)) %>% # 设置标签
add_row(name = "origin", is_main = FALSE) %>% # 添加起点为"origin"的节点
arrange(group, name) # 按组和名称排序

nodes$id <- NA # 新建一个id列，并将值初始化为NA
is_leaf <- nodes$group != "origin" & nodes$name != "origin" # 判断是否为叶子节点
nleaves <- nrow(nodes[is_leaf, ]) # 计算叶子节点数目
nodes$id[is_leaf] <- seq_len(nleaves) # 为叶子节点赋值id
nodes$angle <- 90 - 360 * nodes$id / nleaves # 计算角度

nodes$hjust <- ifelse(nodes$angle < -90, 1, 0) # 设置水平方向对齐方式
nodes$angle <- ifelse(nodes$angle < -90, nodes$angle + 180, nodes$angle) # 调整角度
nodes <- nodes %>%
mutate(angle = replace_na(angle, 0), # 将NA替换为0
hjust = replace_na(hjust, 0))
```
### 根据节点和边文件创建一个igraph对象
```r
graph <- igraph::graph_from_data_frame(edges, vertices = nodes)
```
### 数据可视化
```r
ggraph(graph, layout = "dendrogram", circular = TRUE) +
  geom_edge_diagonal(color = "grey70", edge_width = 0.33) +
  geom_node_point(
    aes(fill = group, size = size), 
    shape = 21, color = "white", stroke = 0.2, alpha = 0.8,
    show.legend = TRUE) +
  geom_node_text(
    aes(filter = !is_main, label = label, angle = angle, hjust = hjust,
        fontface = ifelse(is_main, "bold", "plain")),
    size = 3.5,
  ) +
  geom_node_text(
    aes(filter = is_main, label = label, angle = angle, hjust = hjust,
        fontface = "bold"),size = 3.5, repel = TRUE) +
  scale_edge_color_manual(values = c(flora_palette, "grey40", "grey80")) +
  scale_fill_manual(values = c(flora_palette, "grey40", "grey80")) +
  scale_size_area(max_size =20) +
  coord_fixed(clip = "off") +
  guides( color = "none", fill = "none",
    size = guide_legend(override.aes = list(color = "grey50"))) +
  theme_void() +
  theme(
    plot.background = element_rect(color = "white", fill = "white"),
    plot.margin = margin(t = 100, r = 100,b = 100, l = 100),
    legend.position ="non")
```
