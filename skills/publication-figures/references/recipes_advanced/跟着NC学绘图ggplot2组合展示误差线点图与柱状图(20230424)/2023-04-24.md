```
p1 <- df %>% ggplot(aes(mean,group,color=type,fill=type))+
  geom_point(position = position_dodge(0.8))+ # 散点
  geom_errorbar(aes(xmin=mean-error,xmax=mean+error), # 误差线
                position = position_dodge(0.8))+
  labs(y=NULL,x="standardized coefficient")+ # x轴标题
  scale_fill_npg()+ # 配色
  scale_color_npg()+ # 配色
  theme_bw()+ # 主题
  theme(axis.text= element_text(color = "black",size=8,face="bold"),
        axis.title = element_text(color="black",size=9,face="bold"),
        legend.title = element_blank(),
        legend.text = element_text(color="black",size=7,face="bold"),
        legend.position = c(0.2,0.065), # 图例位置
        legend.background = element_blank(), # 图例背景
        legend.spacing.x = unit(0.1,"cm"),
        legend.key = element_blank(),
        legend.key.height = unit(0.4,"cm"),
        legend.key.width = unit(0.4,"cm"),
        plot.background = element_blank(),
        panel.background = element_blank())
```
### 创建柱状图
```
p2 <- df %>% ggplot(aes(`IncMSE (%)`,group,color=type,fill=type))+
  geom_col(position = position_dodge(0.8))+ # 柱状图
  scale_x_continuous(expand = c(0,0))+ # x轴缩放
  labs(y=NULL)+ # y轴标题
  scale_fill_npg()+ # 配色
  scale_color_npg()+ # 配色
  theme_bw()+ # 主题
  theme(axis.text.y=element_blank(),
        axis.text.x=element_text(color="black",size=8),
        axis.title = element_text(color="black",size=9,face="bold"),
        axis.ticks.y=element_blank(), # y轴刻度线
        legend.position = "non", # 图例位置
        plot.background = element_blank(), # 图背景
        panel.background = element_blank()) # 面板背景
```
